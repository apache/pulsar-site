375 // By default, when the client connects to the broker, a version string like "Pulsar-CPP-v<x.y.z>" will be
376 // carried and saved by the broker. The client version string could be queried from the topic stats.
377 //
378 // This method provides a way to add more description to a specific `Client` instance. If it's configured,
379 // the description will be appended to the original client version string, with '-' as the separator.
380 //
381 // For example, if the client version is 3.2.0, and the description is "forked", the final client version
382 // string will be "Pulsar-CPP-v3.2.0-forked".
383 //
384 // NOTE: This method should only be called by the PulsarWrapper and the length should not exceed 64.
385 //
386 // For example, you can add a PulsarWrapper class like:
387 //
388 // ```c++
389 // namespace pulsar {
390 // class PulsarWrapper {
391 // static ClientConfiguration clientConfig() {
392 // ClientConfiguration conf;
393 // conf.setDescription("forked");
394 // return conf;
395 // }
396 // };
397 // }
398 // ```
399 //
400 // Then, call the method before passing the `conf` to the constructor of `Client`:
401 //
402 // ```c++
403 // auto conf = PulsarWrapper::clientConfig();
404 // // Set other attributes of `conf` here...
405 // Client client{"pulsar://localhost:6650", conf);
406 // ```
407 ClientConfiguration& setDescription(const std::string& description);
408
409 const std::string& getDescription() const noexcept;
